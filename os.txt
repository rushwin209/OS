#include <stdio.h>

int main() {
    int n, i;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], ct[n], tat[n], wt[n], pid[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter Arrival Time and Burst Time for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
    }

    // Sorting based on Arrival Time (Bubble Sort)
    for(i = 0; i < n - 1; i++) {
        for(int j = 0; j < n - i - 1; j++) {
            if(at[j] > at[j+1]) {
                // Swap AT
                int temp = at[j]; at[j] = at[j+1]; at[j+1] = temp;
                // Swap BT
                temp = bt[j]; bt[j] = bt[j+1]; bt[j+1] = temp;
                // Swap PID
                temp = pid[j]; pid[j] = pid[j+1]; pid[j+1] = temp;
            }
        }
    }

    // Calculating CT, TAT, WT
    for(i = 0; i < n; i++) {
        if(i == 0 || at[i] > ct[i-1]) {
            ct[i] = at[i] + bt[i];
        } else {
            ct[i] = ct[i-1] + bt[i];
        }
        tat[i] = ct[i] - at[i];
        wt[i] = tat[i] - bt[i];
        
        avg_tat += tat[i];
        avg_wt += wt[i];
    }

    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    // n: number of processes, i: loop counter
    int n, i;
    // avg_tat: average turnaround time, avg_wt: average waiting time
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // at: Arrival, bt: Burst, ct: Completion, tat: Turnaround, wt: Waiting
    // is_completed: keeps track of finished processes (0=no, 1=yes)
    int at[n], bt[n], ct[n], tat[n], wt[n], pid[n], is_completed[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1; // Assigning IDs starting from 1
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        is_completed[i] = 0; // Initially, no process is finished
    }

    int completed = 0; // Tracks how many processes have finished execution
    int current_time = 0; // Simulates the CPU clock starting at 0

    while(completed != n) {
        int index = -1; // Stores the index of the best process to run next
        int min_bt = 10000; // A large value to help find the smallest Burst Time

        // Loop to find the process with the shortest BT that has arrived
        for(i = 0; i < n; i++) {
            // Logic: Process must have arrived AND not be finished yet
            if(at[i] <= current_time && is_completed[i] == 0) {
                // Check if this process is shorter than the current shortest found
                if(bt[i] < min_bt) {
                    min_bt = bt[i];
                    index = i;
                }
                // If two processes have same BT, the one that arrived first wins (FCFS tie-break)
                if(bt[i] == min_bt) {
                    if(at[i] < at[index]) {
                        index = i;
                    }
                }
            }
        }

        if(index != -1) {
            // Process found! Completion Time = Current Clock + Execution Time
            ct[index] = current_time + bt[index];
            // Turnaround Time = Total time in system (Exit - Arrival)
            tat[index] = ct[index] - at[index];
            // Waiting Time = Total time in system - Time actually working
            wt[index] = tat[index] - bt[index];

            avg_tat += tat[index];
            avg_wt += wt[index];
            is_completed[index] = 1; // Mark as done so we don't pick it again
            completed++; // Increment the global counter
            current_time = ct[index]; // Jump the clock to the end of this process
        } else {
            // No process has arrived yet; move the clock forward by 1 (Idle state)
            current_time++;
        }
    }

    // Standard output table
    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    int n, i, completed = 0, current_time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // rt[]: Remaining Time. We need this because BT will change as we work.
    int at[n], bt[n], rt[n], ct[n], tat[n], wt[n], pid[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        rt[i] = bt[i]; // Initially, remaining time is the full burst time
    }

    // We use a large value for comparison to find the minimum
    int min_rt = 10000;
    int shortest = 0;
    int check = 0; // To check if a process was actually found at this time

    // Simulation runs second by second (Preemptive)
    while (completed != n) {
        min_rt = 10000;
        check = 0;

        // Step 1: Find the process with the shortest REMAINING time at this second
        for (i = 0; i < n; i++) {
            if ((at[i] <= current_time) && (rt[i] < min_rt) && rt[i] > 0) {
                min_rt = rt[i];
                shortest = i;
                check = 1;
            }
        }

        if (check == 0) {
            current_time++; // CPU is idle if no process has arrived
            continue;
        }

        // Step 2: Execute the shortest process for just 1 unit of time
        rt[shortest]--;

        // Step 3: If process is finished
        if (rt[shortest] == 0) {
            completed++;
            ct[shortest] = current_time + 1;
            
            // Calculate TAT and WT immediately
            tat[shortest] = ct[shortest] - at[shortest];
            wt[shortest] = tat[shortest] - bt[shortest];
            
            if (wt[shortest] < 0) wt[shortest] = 0; // Avoid negative values
        }
        
        // Move to the next second
        current_time++;
    }

    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        avg_tat += tat[i];
        avg_wt += wt[i];
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    // n: number of processes, i, j: counters, tq: time quantum
    // completed: count of finished jobs, current_time: global clock
    int n, i, j, tq, completed = 0, current_time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // rt[]: Remaining Time (critical for RR as we work in slices)
    // exist[]: Tracks if a process is already in the ready queue to avoid duplicates
    int at[n], bt[n], rt[n], ct[n], tat[n], wt[n], pid[n], exist[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        rt[i] = bt[i];
        exist[i] = 0; // 0 means not in queue
    }

    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    // Queue logic for RR
    int queue[100]; // Simple array to act as a FIFO queue
    int front = 0, rear = 0;

    // Sort by AT first so we know who arrives at Time 0
    for(i = 0; i < n - 1; i++) {
        for(j = 0; j < n - i - 1; j++) {
            if(at[j] > at[j+1]) {
                int temp;
                temp = at[j]; at[j] = at[j+1]; at[j+1] = temp;
                temp = bt[j]; bt[j] = bt[j+1]; bt[j+1] = temp;
                temp = rt[j]; rt[j] = rt[j+1]; rt[j+1] = temp;
                temp = pid[j]; pid[j] = pid[j+1]; pid[j+1] = temp;
            }
        }
    }

    // Push the first process that arrives into the queue
    queue[rear++] = 0; 
    exist[0] = 1;

    while(completed != n) {
        // Get the process from the front of the queue
        int idx = queue[front++];
        
        // Scenario A: Process needs more time than the quantum
        if(rt[idx] > tq) {
            rt[idx] -= tq;
            current_time += tq;
        } 
        // Scenario B: Process finishes within this quantum
        else {
            current_time += rt[idx];
            rt[idx] = 0;
            completed++;
            ct[idx] = current_time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - bt[idx];
            avg_tat += tat[idx];
            avg_wt += wt[idx];
        }

        // Check for newly arrived processes while the current one was running
        for(i = 0; i < n; i++) {
            if(at[i] <= current_time && exist[i] == 0 && rt[i] > 0) {
                queue[rear++] = i;
                exist[i] = 1;
            }
        }

        // If the current process isn't finished, it goes to the back of the queue
        if(rt[idx] > 0) {
            queue[rear++] = idx;
        }

        // If the queue is empty but we aren't done, skip time to next arrival
        if(front == rear && completed < n) {
            for(i = 0; i < n; i++) {
                if(rt[i] > 0) {
                    current_time = at[i];
                    queue[rear++] = i;
                    exist[i] = 1;
                    break;
                }
            }
        }
    }

    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    int n, i, completed = 0, current_time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // pid[] acts as the Priority (Lower PID = Higher Priority)
    int at[n], bt[n], ct[n], tat[n], wt[n], pid[n], is_completed[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1; // Process ID 1, 2, 3...
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        is_completed[i] = 0;
    }

    while(completed != n) {
        int best_prio = 10000; // High value to find the minimum PID
        int index = -1;

        for(i = 0; i < n; i++) {
            // Check if arrived, not finished, and has a smaller PID (Higher Priority)
            if(at[i] <= current_time && is_completed[i] == 0) {
                if(pid[i] < best_prio) {
                    best_prio = pid[i];
                    index = i;
                }
            }
        }

        if(index != -1) {
            // Non-preemptive: Run the whole burst at once
            ct[index] = current_time + bt[index];
            tat[index] = ct[index] - at[index];
            wt[index] = tat[index] - bt[index];
            avg_tat += tat[index];
            avg_wt += wt[index];
            is_completed[index] = 1;
            completed++;
            current_time = ct[index]; // Jump clock to end of process
        } else {
            current_time++; // No one has arrived yet
        }
    }
    // Print logic (similar to previous programs)
    printf("\nPID(Prio)\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) printf("%d\t\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    return 0;
}

                wt[shortest] = tat[shortest] - bt[shortest];

                avg_wt += wt[shortest];

        current_time++; // Clock ticks regardless of work
    }

    printf("\nPID(Prio)\tAT\tBT\tCT\tTAT\tWT\n");

}

#include <stdio.h>

int main() {
    // n: Processes, m: Resource types
    int n, m, i, j, k;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    printf("Enter number of resource types: ");
    scanf("%d", &m);

    // alloc[n][m]: Resources currently assigned to processes
    // max[n][m]: Maximum resources a process might request
    // avail[m]: Resources currently free in the system
    int alloc[n][m], max[n][m], avail[m];

    printf("Enter Allocation Matrix:\n");
    for (i = 0; i < n; i++)
        for (j = 0; j < m; j++) scanf("%d", &alloc[i][j]);

    printf("Enter Max Matrix:\n");
    for (i = 0; i < n; i++)
        for (j = 0; j < m; j++) scanf("%d", &max[i][j]);

    printf("Enter Available Resources:\n");
    for (i = 0; i < m; i++) scanf("%d", &avail[i]);

    // finish[n]: Boolean-like array to track if a process has finished its execution
    // ans[n]: Stores the resulting safe sequence of process IDs
    // ind: Counter for the safe sequence array
    int finish[n], ans[n], ind = 0;
    for (k = 0; k < n; k++) {
        finish[k] = 0; // Initialize all processes as NOT finished
    }

    // need[n][m]: Remaining resources required by each process to finish
    // Calculated as: Need = Max - Allocation
    int need[n][m];
    for (i = 0; i < n; i++) {
        for (j = 0; j < m; j++)
            need[i][j] = max[i][j] - alloc[i][j];
    }

    // Work[]: Temporary copy of Available to simulate resource reclamation
    int work[m];
    for (i = 0; i < m; i++) {
        work[i] = avail[i];
    }

    // The Safety Algorithm
    for (k = 0; k < n; k++) { // We need to check all processes
        for (i = 0; i < n; i++) { // Loop through each process
            if (finish[i] == 0) { // If process 'i' is not yet finished
                int flag = 0; // flag to check if Need <= Work
                for (j = 0; j < m; j++) {
                    if (need[i][j] > work[j]) {
                        flag = 1; // Process cannot finish yet
                        break;
                    }
                }

                if (flag == 0) { // If all resource needs can be met
                    ans[ind++] = i; // Add to Safe Sequence
                    for (int y = 0; y < m; y++)
                        work[y] += alloc[i][y]; // Reclaim resources after finish
                    finish[i] = 1; // Mark as finished
                }
            }
        }
    }

    // Checking if all processes were finished successfully
    int safe = 1;
    for(i = 0; i < n; i++) {
        if(finish[i] == 0) {
            safe = 0; // If any process is not finished, system is unsafe
            printf("\nSystem is NOT in a safe state!");
            break;
        }
    }

    if(safe) {
        printf("\nSystem is in a SAFE State.\nSafe Sequence: ");
        for (i = 0; i < n - 1; i++)
            printf(" P%d ->", ans[i]);
        printf(" P%d\n", ans[n - 1]);
    }

    return (0);
}

#include <stdio.h>

int main() {
    // ms: memory size, ps: page size, nof: number of frames
    // np: number of pages, remp: remaining pages
    int ms, ps, nof, np, remp, i, j;
    
    // s[20]: page table array where index is Page No and value is Frame No
    // la: logical address, pa: physical address, pno: page number, off: offset
    int s[20], la, pa, pno, off;

    printf("Enter the memory size: ");
    scanf("%d", &ms);

    printf("Enter the page size: ");
    scanf("%d", &ps);

    // Calculating total frames available in physical memory
    nof = ms / ps;
    printf("The no. of frames available in memory are: %d", nof);

    printf("\nEnter number of pages for the process: ");
    scanf("%d", &np);

    // remp helps track if we have enough physical frames to accommodate the process
    remp = nof;
    if (np > remp) {
        printf("\nMemory is Full! Cannot allocate process.");
        return 0;
    }

    printf("\nEnter relative mapping (Page Table):\n");
    for (i = 0; i < np; i++) {
        printf("Enter frame no for page %d: ", i);
        // The page table maps Page 'i' to a specific Frame Number
        scanf("%d", &s[i]);
    }

    // Logical Address translation simulation
    printf("\nEnter Logical Address (Page Number and Offset): ");
    // Logical address is often represented as (p, d)
    scanf("%d %d", &pno, &off);

    // Validation: Check if the page number exists within the process
    if (pno >= np || pno < 0) {
        printf("\nInvalid Page Number!");
    }
    // Validation: Offset must be smaller than the page size
    else if (off >= ps || off < 0) {
        printf("\nInvalid Offset (Offset must be less than Page Size %d)!", ps);
    }
    else {
        // Physical Address Calculation Logic:
        // Physical Address = (Frame Number * Page Size) + Offset
        // We look up the Frame Number in our Page Table array 's'
        pa = (s[pno] * ps) + off;
        
        printf("\nThe Physical Address is: %d\n", pa);
    }

    return 0;
}

#include <stdio.h>

int main() {
    // n: total segments, i: counter
    int n, i;
    
    // base[10]: stores the starting physical address of each segment
    // limit[10]: stores the maximum size (length) of each segment
    int base[10], limit[10];
    
    // seg: requested segment, off: requested offset, pa: resulting physical address
    int seg, off, pa;

    printf("Enter the number of segments: ");
    scanf("%d", &n);

    // Initializing the Segment Table
    for (i = 0; i < n; i++) {
        printf("\nEnter details for Segment %d:", i);
        printf("\nEnter Base Address: ");
        scanf("%d", &base[i]);
        printf("Enter Limit: ");
        scanf("%d", &limit[i]);
    }

    // Address Translation Request
    printf("\nEnter Logical Address to find Physical Address");
    printf("\nEnter Segment Number: ");
    scanf("%d", &seg);
    printf("Enter Offset: ");
    scanf("%d", &off);

    // Step 1: Check if the segment index exists in the table
    if (seg >= n || seg < 0) {
        printf("\nInvalid Segment Number!");
    } 
    // Step 2: Boundary Check (The Limit Check)
    // The offset must be strictly LESS than the limit (0 to limit-1)
    else if (off < limit[seg]) {
        // Physical Address = Base of the segment + the relative offset
        pa = base[seg] + off;
        printf("\nLogical Address is valid.");
        printf("\nPhysical Address is: %d\n", pa);
    } 
    else {
        // This is known as a "Segmentation Fault" or Trap in a real OS
        printf("\nAddressing Trap: Offset %d exceeds Segment Limit %d!", off, limit[seg]);
    }

    return 0;
}
