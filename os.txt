#include <stdio.h>

int main() {
    int n, i;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], ct[n], tat[n], wt[n], pid[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter Arrival Time and Burst Time for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
    }

    // Sorting based on Arrival Time (Bubble Sort)
    for(i = 0; i < n - 1; i++) {
        for(int j = 0; j < n - i - 1; j++) {
            if(at[j] > at[j+1]) {
                // Swap AT
                int temp = at[j]; at[j] = at[j+1]; at[j+1] = temp;
                // Swap BT
                temp = bt[j]; bt[j] = bt[j+1]; bt[j+1] = temp;
                // Swap PID
                temp = pid[j]; pid[j] = pid[j+1]; pid[j+1] = temp;
            }
        }
    }

    // Calculating CT, TAT, WT
    for(i = 0; i < n; i++) {
        if(i == 0 || at[i] > ct[i-1]) {
            ct[i] = at[i] + bt[i];
        } else {
            ct[i] = ct[i-1] + bt[i];
        }
        tat[i] = ct[i] - at[i];
        wt[i] = tat[i] - bt[i];
        
        avg_tat += tat[i];
        avg_wt += wt[i];
    }

    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    // n: number of processes, i: loop counter
    int n, i;
    // avg_tat: average turnaround time, avg_wt: average waiting time
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // at: Arrival, bt: Burst, ct: Completion, tat: Turnaround, wt: Waiting
    // is_completed: keeps track of finished processes (0=no, 1=yes)
    int at[n], bt[n], ct[n], tat[n], wt[n], pid[n], is_completed[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1; // Assigning IDs starting from 1
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        is_completed[i] = 0; // Initially, no process is finished
    }

    int completed = 0; // Tracks how many processes have finished execution
    int current_time = 0; // Simulates the CPU clock starting at 0

    while(completed != n) {
        int index = -1; // Stores the index of the best process to run next
        int min_bt = 10000; // A large value to help find the smallest Burst Time

        // Loop to find the process with the shortest BT that has arrived
        for(i = 0; i < n; i++) {
            // Logic: Process must have arrived AND not be finished yet
            if(at[i] <= current_time && is_completed[i] == 0) {
                // Check if this process is shorter than the current shortest found
                if(bt[i] < min_bt) {
                    min_bt = bt[i];
                    index = i;
                }
                // If two processes have same BT, the one that arrived first wins (FCFS tie-break)
                if(bt[i] == min_bt) {
                    if(at[i] < at[index]) {
                        index = i;
                    }
                }
            }
        }

        if(index != -1) {
            // Process found! Completion Time = Current Clock + Execution Time
            ct[index] = current_time + bt[index];
            // Turnaround Time = Total time in system (Exit - Arrival)
            tat[index] = ct[index] - at[index];
            // Waiting Time = Total time in system - Time actually working
            wt[index] = tat[index] - bt[index];

            avg_tat += tat[index];
            avg_wt += wt[index];
            is_completed[index] = 1; // Mark as done so we don't pick it again
            completed++; // Increment the global counter
            current_time = ct[index]; // Jump the clock to the end of this process
        } else {
            // No process has arrived yet; move the clock forward by 1 (Idle state)
            current_time++;
        }
    }

    // Standard output table
    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    int n, i, completed = 0, current_time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // rt[]: Remaining Time. We need this because BT will change as we work.
    int at[n], bt[n], rt[n], ct[n], tat[n], wt[n], pid[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        rt[i] = bt[i]; // Initially, remaining time is the full burst time
    }

    // We use a large value for comparison to find the minimum
    int min_rt = 10000;
    int shortest = 0;
    int check = 0; // To check if a process was actually found at this time

    // Simulation runs second by second (Preemptive)
    while (completed != n) {
        min_rt = 10000;
        check = 0;

        // Step 1: Find the process with the shortest REMAINING time at this second
        for (i = 0; i < n; i++) {
            if ((at[i] <= current_time) && (rt[i] < min_rt) && rt[i] > 0) {
                min_rt = rt[i];
                shortest = i;
                check = 1;
            }
        }

        if (check == 0) {
            current_time++; // CPU is idle if no process has arrived
            continue;
        }

        // Step 2: Execute the shortest process for just 1 unit of time
        rt[shortest]--;

        // Step 3: If process is finished
        if (rt[shortest] == 0) {
            completed++;
            ct[shortest] = current_time + 1;
            
            // Calculate TAT and WT immediately
            tat[shortest] = ct[shortest] - at[shortest];
            wt[shortest] = tat[shortest] - bt[shortest];
            
            if (wt[shortest] < 0) wt[shortest] = 0; // Avoid negative values
        }
        
        // Move to the next second
        current_time++;
    }

    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        avg_tat += tat[i];
        avg_wt += wt[i];
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

#include <stdio.h>

int main() {
    // n: number of processes, i, j: counters, tq: time quantum
    // completed: count of finished jobs, current_time: global clock
    int n, i, j, tq, completed = 0, current_time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // rt[]: Remaining Time (critical for RR as we work in slices)
    // exist[]: Tracks if a process is already in the ready queue to avoid duplicates
    int at[n], bt[n], rt[n], ct[n], tat[n], wt[n], pid[n], exist[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1;
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        rt[i] = bt[i];
        exist[i] = 0; // 0 means not in queue
    }

    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    // Queue logic for RR
    int queue[100]; // Simple array to act as a FIFO queue
    int front = 0, rear = 0;

    // Sort by AT first so we know who arrives at Time 0
    for(i = 0; i < n - 1; i++) {
        for(j = 0; j < n - i - 1; j++) {
            if(at[j] > at[j+1]) {
                int temp;
                temp = at[j]; at[j] = at[j+1]; at[j+1] = temp;
                temp = bt[j]; bt[j] = bt[j+1]; bt[j+1] = temp;
                temp = rt[j]; rt[j] = rt[j+1]; rt[j+1] = temp;
                temp = pid[j]; pid[j] = pid[j+1]; pid[j+1] = temp;
            }
        }
    }

    // Push the first process that arrives into the queue
    queue[rear++] = 0; 
    exist[0] = 1;

    while(completed != n) {
        // Get the process from the front of the queue
        int idx = queue[front++];
        
        // Scenario A: Process needs more time than the quantum
        if(rt[idx] > tq) {
            rt[idx] -= tq;
            current_time += tq;
        } 
        // Scenario B: Process finishes within this quantum
        else {
            current_time += rt[idx];
            rt[idx] = 0;
            completed++;
            ct[idx] = current_time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - bt[idx];
            avg_tat += tat[idx];
            avg_wt += wt[idx];
        }

        // Check for newly arrived processes while the current one was running
        for(i = 0; i < n; i++) {
            if(at[i] <= current_time && exist[i] == 0 && rt[i] > 0) {
                queue[rear++] = i;
                exist[i] = 1;
            }
        }

        // If the current process isn't finished, it goes to the back of the queue
        if(rt[idx] > 0) {
            queue[rear++] = idx;
        }

        // If the queue is empty but we aren't done, skip time to next arrival
        if(front == rear && completed < n) {
            for(i = 0; i < n; i++) {
                if(rt[i] > 0) {
                    current_time = at[i];
                    queue[rear++] = i;
                    exist[i] = 1;
                    break;
                }
            }
        }
    }

    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f", avg_tat/n);
    printf("\nAverage Waiting Time: %.2f\n", avg_wt/n);

    return 0;
}

int main() {
    int n, i, completed = 0, current_time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    // pid[] acts as the Priority (Lower PID = Higher Priority)
    int at[n], bt[n], ct[n], tat[n], wt[n], pid[n], is_completed[n];

    for(i = 0; i < n; i++) {
        pid[i] = i + 1; // Process ID 1, 2, 3...
        printf("Enter AT and BT for Process %d: ", pid[i]);
        scanf("%d %d", &at[i], &bt[i]);
        is_completed[i] = 0;
    }

    while(completed != n) {
        int best_prio = 10000; // High value to find the minimum PID
        int index = -1;

        for(i = 0; i < n; i++) {
            // Check if arrived, not finished, and has a smaller PID (Higher Priority)
            if(at[i] <= current_time && is_completed[i] == 0) {
                if(pid[i] < best_prio) {
                    best_prio = pid[i];
                    index = i;
                }
            }
        }

        if(index != -1) {
            // Non-preemptive: Run the whole burst at once
            ct[index] = current_time + bt[index];
            tat[index] = ct[index] - at[index];
            wt[index] = tat[index] - bt[index];
            avg_tat += tat[index];
            avg_wt += wt[index];
            is_completed[index] = 1;
            completed++;
            current_time = ct[index]; // Jump clock to end of process
        } else {
            current_time++; // No one has arrived yet
        }
    }
    // Print logic (similar to previous programs)
    printf("\nPID(Prio)\tAT\tBT\tCT\tTAT\tWT\n");
    for(i = 0; i < n; i++) printf("%d\t\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    return 0;
}

#include <stdio.h>

int main() {
    // n: Processes, m: Resource types
    int n, m, i, j, k;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    printf("Enter number of resource types: ");
    scanf("%d", &m);

    // alloc[n][m]: Resources currently assigned to processes
    // max[n][m]: Maximum resources a process might request
    // avail[m]: Resources currently free in the system
    int alloc[n][m], max[n][m], avail[m];

    printf("Enter Allocation Matrix:\n");
    for (i = 0; i < n; i++)
        for (j = 0; j < m; j++) scanf("%d", &alloc[i][j]);

    printf("Enter Max Matrix:\n");
    for (i = 0; i < n; i++)
        for (j = 0; j < m; j++) scanf("%d", &max[i][j]);

    printf("Enter Available Resources:\n");
    for (i = 0; i < m; i++) scanf("%d", &avail[i]);

    // finish[n]: Boolean-like array to track if a process has finished its execution
    // ans[n]: Stores the resulting safe sequence of process IDs
    // ind: Counter for the safe sequence array
    int finish[n], ans[n], ind = 0;
    for (k = 0; k < n; k++) {
        finish[k] = 0; // Initialize all processes as NOT finished
    }

    // need[n][m]: Remaining resources required by each process to finish
    // Calculated as: Need = Max - Allocation
    int need[n][m];
    for (i = 0; i < n; i++) {
        for (j = 0; j < m; j++)
            need[i][j] = max[i][j] - alloc[i][j];
    }

    // Work[]: Temporary copy of Available to simulate resource reclamation
    int work[m];
    for (i = 0; i < m; i++) {
        work[i] = avail[i];
    }

    // The Safety Algorithm
    for (k = 0; k < n; k++) { // We need to check all processes
        for (i = 0; i < n; i++) { // Loop through each process
            if (finish[i] == 0) { // If process 'i' is not yet finished
                int flag = 0; // flag to check if Need <= Work
                for (j = 0; j < m; j++) {
                    if (need[i][j] > work[j]) {
                        flag = 1; // Process cannot finish yet
                        break;
                    }
                }

                if (flag == 0) { // If all resource needs can be met
                    ans[ind++] = i; // Add to Safe Sequence
                    for (int y = 0; y < m; y++)
                        work[y] += alloc[i][y]; // Reclaim resources after finish
                    finish[i] = 1; // Mark as finished
                }
            }
        }
    }

    // Checking if all processes were finished successfully
    int safe = 1;
    for(i = 0; i < n; i++) {
        if(finish[i] == 0) {
            safe = 0; // If any process is not finished, system is unsafe
            printf("\nSystem is NOT in a safe state!");
            break;
        }
    }

    if(safe) {
        printf("\nSystem is in a SAFE State.\nSafe Sequence: ");
        for (i = 0; i < n - 1; i++)
            printf(" P%d ->", ans[i]);
        printf(" P%d\n", ans[n - 1]);
    }

    return (0);
}

#include <stdio.h>

int main() {
    // ms: memory size, ps: page size, nof: number of frames
    // np: number of pages, remp: remaining pages
    int ms, ps, nof, np, remp, i, j;
    
    // s[20]: page table array where index is Page No and value is Frame No
    // la: logical address, pa: physical address, pno: page number, off: offset
    int s[20], la, pa, pno, off;

    printf("Enter the memory size: ");
    scanf("%d", &ms);

    printf("Enter the page size: ");
    scanf("%d", &ps);

    // Calculating total frames available in physical memory
    nof = ms / ps;
    printf("The no. of frames available in memory are: %d", nof);

    printf("\nEnter number of pages for the process: ");
    scanf("%d", &np);

    // remp helps track if we have enough physical frames to accommodate the process
    remp = nof;
    if (np > remp) {
        printf("\nMemory is Full! Cannot allocate process.");
        return 0;
    }

    printf("\nEnter relative mapping (Page Table):\n");
    for (i = 0; i < np; i++) {
        printf("Enter frame no for page %d: ", i);
        // The page table maps Page 'i' to a specific Frame Number
        scanf("%d", &s[i]);
    }

    // Logical Address translation simulation
    printf("\nEnter Logical Address (Page Number and Offset): ");
    // Logical address is often represented as (p, d)
    scanf("%d %d", &pno, &off);

    // Validation: Check if the page number exists within the process
    if (pno >= np || pno < 0) {
        printf("\nInvalid Page Number!");
    }
    // Validation: Offset must be smaller than the page size
    else if (off >= ps || off < 0) {
        printf("\nInvalid Offset (Offset must be less than Page Size %d)!", ps);
    }
    else {
        // Physical Address Calculation Logic:
        // Physical Address = (Frame Number * Page Size) + Offset
        // We look up the Frame Number in our Page Table array 's'
        pa = (s[pno] * ps) + off;
        
        printf("\nThe Physical Address is: %d\n", pa);
    }

    return 0;
}

#include <stdio.h>

int main() {
    // n: total segments, i: counter
    int n, i;
    
    // base[10]: stores the starting physical address of each segment
    // limit[10]: stores the maximum size (length) of each segment
    int base[10], limit[10];
    
    // seg: requested segment, off: requested offset, pa: resulting physical address
    int seg, off, pa;

    printf("Enter the number of segments: ");
    scanf("%d", &n);

    // Initializing the Segment Table
    for (i = 0; i < n; i++) {
        printf("\nEnter details for Segment %d:", i);
        printf("\nEnter Base Address: ");
        scanf("%d", &base[i]);
        printf("Enter Limit: ");
        scanf("%d", &limit[i]);
    }

    // Address Translation Request
    printf("\nEnter Logical Address to find Physical Address");
    printf("\nEnter Segment Number: ");
    scanf("%d", &seg);
    printf("Enter Offset: ");
    scanf("%d", &off);

    // Step 1: Check if the segment index exists in the table
    if (seg >= n || seg < 0) {
        printf("\nInvalid Segment Number!");
    } 
    // Step 2: Boundary Check (The Limit Check)
    // The offset must be strictly LESS than the limit (0 to limit-1)
    else if (off < limit[seg]) {
        // Physical Address = Base of the segment + the relative offset
        pa = base[seg] + off;
        printf("\nLogical Address is valid.");
        printf("\nPhysical Address is: %d\n", pa);
    } 
    else {
        // This is known as a "Segmentation Fault" or Trap in a real OS
        printf("\nAddressing Trap: Offset %d exceeds Segment Limit %d!", off, limit[seg]);
    }

    return 0;
}

    // f: Number of frames (RAM capacity)
    // fault: Counter for page faults
    // pointer: Index to keep track of the oldest page (FIFO queue)
    int i, j, n, f, fault = 0, pointer = 0;

    printf("Enter number of frames: ");
    scanf("%d", &f);

    // frames[f]: Array representing physical memory slots
    int frames[f];
    for (i = 0; i < f; i++) {
        frames[i] = -1; // Initialize frames with -1 (meaning empty)
    }

    printf("Enter number of pages in reference string: ");
    scanf("%d", &n);

    // ref[n]: Array to store the sequence of page requests
    int ref[n];
    printf("Enter the reference string: ");
    for (i = 0; i < n; i++) {
        scanf("%d", &ref[i]);
    }

    printf("\nString | Frame Contents | Status");
    printf("\n--------------------------------");

    for (i = 0; i < n; i++) {
        printf("\n  %d    |", ref[i]);
        
        // flag: Used to check if the page is already in memory (Hit = 1, Miss = 0)
        int flag = 0;
        for (j = 0; j < f; j++) {
            if (frames[j] == ref[i]) {
                flag = 1; // Page Hit
                break;
            }
        }

        // If flag remains 0, we have a Page Fault
        if (flag == 0) {
            // Replace the oldest page using the pointer
            frames[pointer] = ref[i];
            
            // Circular increment of pointer: 0 -> 1 -> 2 -> 0...
            // This ensures we always point to the oldest entry
            pointer = (pointer + 1) % f;
            
            fault++; // Increment total faults

            // Printing the current frames after a fault
            for (k = 0; k < f; k++) {
                if (frames[k] != -1) printf(" %d", frames[k]);
                else printf(" -");
            }
            printf(" | Fault");
        } else {
            // Printing current frames after a hit
            for (k = 0; k < f; k++) {
                if (frames[k] != -1) printf(" %d", frames[k]);
                else printf(" -");
            }
            printf(" | Hit");
        }
    }

    printf("\n\nTotal Page Faults: %d\n", fault);

    return 0;
}
Page replacement LRU
--------------------------------------------------------
#include <stdio.h>

// findLRU: Function to find the index of the page that was used furthest in the past
int findLRU(int time[], int n) {
    int i, minimum = time[0], pos = 0;
    for (i = 1; i < n; i++) {
        // The page with the lowest counter value is the Least Recently Used
        if (time[i] < minimum) {
            minimum = time[i];
            pos = i;
        }
    }
    return pos;
}

int main() {
    // f: total frames, n: length of reference string, fault: total misses
    // counter: a global 'clock' that increments with every request
    int f, n, i, j, fault = 0, counter = 0;

    printf("Enter number of frames: ");
    scanf("%d", &f);

    printf("Enter number of pages in reference string: ");
    scanf("%d", &n);

    // frames[]: stores the page numbers currently in RAM
    // time[]: stores the 'time-stamp' of when each frame was last used
    // ref[]: the sequence of page requests
    int frames[f], time[f], ref[n];

    printf("Enter the reference string: ");
    for (i = 0; i < n; i++) {
        scanf("%d", &ref[i]);
    }

    // Initialize all frames as empty (-1)
    for (i = 0; i < f; i++) {
        frames[i] = -1;
    }

    printf("\nString | Frame Contents | Status");
    printf("\n--------------------------------");

    for (i = 0; i < n; i++) {
        counter++; // Increment the time-stamp clock
        int flag1 = 0, flag2 = 0;

        // Check if page already exists in one of the frames (Hit)
        for (j = 0; j < f; j++) {
            if (frames[j] == ref[i]) {
                flag1 = flag2 = 1; // Mark as Hit
                time[j] = counter; // Update time-stamp because it was just used
                break;
            }
        }

        // If it's a Miss, check if there is an empty frame available
        if (flag1 == 0) {
            for (j = 0; j < f; j++) {
                if (frames[j] == -1) {
                    fault++;
                    frames[j] = ref[i];
                    time[j] = counter; // Set initial time-stamp
                    flag2 = 1;
                    break;
                }
            }
        }

        // If it's a Miss and RAM is full, we must replace the LRU page
        if (flag2 == 0) {
            int pos = findLRU(time, f); // Find oldest time-stamp
            fault++;
            frames[pos] = ref[i]; // Replace it
            time[pos] = counter;  // Update the new page's time-stamp
        }

        // Print Current Status
        printf("\n  %d    |", ref[i]);
        for (j = 0; j < f; j++) {
            if (frames[j] != -1) printf(" %d", frames[j]);
            else printf(" -");
        }
        if (flag1 == 1) printf(" | Hit");
        else printf(" | Fault");
    }

    printf("\n\nTotal Page Faults: %d\n", fault);

    return 0;
}
Paging OPtimal
----------------------------------------------------------
#include <stdio.h>

int main() {
    // n: length of reference string, f: number of frames
    // i, j, k: counters, fault: total page misses
    int n, f, i, j, k, fault = 0;

    printf("Enter number of frames: ");
    scanf("%d", &f);

    printf("Enter number of pages in reference string: ");
    scanf("%d", &n);

    // frames[]: current pages in RAM
    // ref[]: the incoming page request sequence
    int frames[f], ref[n];

    printf("Enter the reference string: ");
    for (i = 0; i < n; i++) {
        scanf("%d", &ref[i]);
    }

    // Initializing frames to -1 (empty)
    for (i = 0; i < f; i++) {
        frames[i] = -1;
    }

    printf("\nString | Frame Contents | Status");
    printf("\n--------------------------------");

    for (i = 0; i < n; i++) {
        int flag1 = 0, flag2 = 0;

        // Check if current page is already in a frame (Hit)
        for (j = 0; j < f; j++) {
            if (frames[j] == ref[i]) {
                flag1 = flag2 = 1;
                break;
            }
        }

        // If Miss, check if there is an empty frame
        if (flag1 == 0) {
            for (j = 0; j < f; j++) {
                if (frames[j] == -1) {
                    fault++;
                    frames[j] = ref[i];
                    flag2 = 1;
                    break;
                }
            }
        }

        // If Miss and frames are full: Optimal Replacement Logic
        if (flag2 == 0) {
            fault++;
            int pos = -1, furthest = i + 1;

            // For each frame, look into the future to see when it will be used again
            for (j = 0; j < f; j++) {
                int next_use = -1;
                for (k = i + 1; k < n; k++) {
                    if (frames[j] == ref[k]) {
                        next_use = k;
                        break;
                    }
                }

                // If a page in the frame is NEVER used again, replace it immediately
                if (next_use == -1) {
                    pos = j;
                    break;
                }

                // Otherwise, keep track of which page is used furthest in the future
                if (next_use > furthest) {
                    furthest = next_use;
                    pos = j;
                }
            }
            
            // If all future uses are tied or not found, default to replacing the first frame
            if (pos == -1) pos = 0;
            
            frames[pos] = ref[i];
        }

        // Display current frames
        printf("\n  %d    |", ref[i]);
        for (j = 0; j < f; j++) {
            if (frames[j] != -1) printf(" %d", frames[j]);
            else printf(" -");
        }
        if (flag1 == 1) printf(" | Hit");
        else printf(" | Fault");
    }

    printf("\n\nTotal Page Faults: %d\n", fault);

    return 0;
}
IPC pipes
-------------------------------------------------------------------------------------
#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <string.h>
#include <stdlib.h>

int main() {
    // fd[2]: File descriptors array. 
    // fd[0] is for READING, fd[1] is for WRITING
    int fd[2];
    
    // pid: To store the process ID after forking
    pid_t pid;
    
    // write_msg: The data we want to send
    // read_msg: The buffer to hold data received from the pipe
    char write_msg[100] = "Hello from the Parent process!";
    char read_msg[100];

    // pipe(fd): System call to create the pipe. 
    // It returns 0 on success and -1 on failure.
    if (pipe(fd) == -1) {
        fprintf(stderr, "Pipe failed");
        return 1;
    }

    // fork(): Creates a child process. 
    // It returns the child's PID to the parent and 0 to the child.
    pid = fork();

    if (pid < 0) { // Error handling for fork
        fprintf(stderr, "Fork failed");
        return 1;
    }

    if (pid > 0) { 
        // --- PARENT PROCESS (Writer) ---
        
        // Always close the end of the pipe you are not using.
        // Parent writes, so we close the READ end (fd[0]).
        close(fd[0]);

        // write(): Writes the message into the pipe's buffer.
        // strlen(write_msg) + 1 includes the null terminator '\0'
        write(fd[1], write_msg, strlen(write_msg) + 1);

        // Close the WRITE end after finishing.
        close(fd[1]);
        printf("Parent: Message sent to pipe.\n");
    } 
    else {
        // --- CHILD PROCESS (Reader) ---
        
        // Child reads, so we close the unused WRITE end (fd[1]).
        close(fd[1]);

        // read(): Suspends child until data is available in the pipe.
        // It reads from fd[0] and stores it in read_msg.
        read(fd[0], read_msg, sizeof(read_msg));

        printf("Child: Received message: %s\n", read_msg);

        // Close the READ end after finishing.
        close(fd[0]);
    }

    return 0;
}
IPC FIFO
------------------------------------------------------
#include <stdio.h>
#include <string.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <unistd.h>

int main() {
    // fd: File descriptor for the FIFO file
    int fd;

    // myfifo: The path to the named pipe file in the file system
    char *myfifo = "/tmp/myfifo";

    // mkfifo: Creates the named pipe. 
    // 0666: Permissions (Owner, Group, Others can Read/Write)
    mkfifo(myfifo, 0666);

    char arr1[80], arr2[80];
    
    // fork() to simulate two different processes using the same FIFO
    pid_t pid = fork();

    if (pid > 0) {
        // --- WRITER PROCESS ---
        
        // Open the FIFO for Write only
        fd = open(myfifo, O_WRONLY);

        printf("Writer: Enter the message: ");
        fgets(arr2, 80, stdin);

        // write(): Sends the string into the FIFO file
        write(fd, arr2, strlen(arr2) + 1);
        
        close(fd);
        printf("Writer: Data sent to FIFO.\n");
    } 
    else {
        // --- READER PROCESS ---
        
        // Open the FIFO for Read only. 
        // Note: open() will block here until the Writer also opens the file.
        fd = open(myfifo, O_RDONLY);

        // read(): Retrieves data from the FIFO
        read(fd, arr1, sizeof(arr1));

        // Print the read string
        printf("Reader: Received message: %s", arr1);
        
        close(fd);

        // unlink(): Deletes the FIFO file from the system after use
        unlink(myfifo);
    }

    return 0;
}
IPC message Queues
-----------------------------------------------------------
#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <string.h>
#include <unistd.h>
#include <stdlib.h>

// struct msg_buffer: The standard structure required for message queues
// The first member MUST be a long (mtype) to specify message priority/category
struct msg_buffer {
    long msg_type;
    char msg_text[100];
} message;

int main() {
    key_t key;
    int msgid;

    // ftok: File to Key. It creates a unique key based on an existing file 
    // and a project ID ('a'). Both processes must use the same key.
    key = ftok("progfile", 'a');

    // msgget: Creates a message queue and returns the identifier
    // 0666: Permissions, IPC_CREAT: Create queue if it doesn't exist
    msgid = msgget(key, 0666 | IPC_CREAT);

    if (fork() > 0) {
        // --- SENDER PROCESS ---
        
        // mtype must be greater than 0
        message.msg_type = 1; 

        printf("Sender: Enter a message to put in queue: ");
        fgets(message.msg_text, 100, stdin);

        // msgsnd: Sends the message to the queue.
        // The size excludes the long mtype (sizeof(message) - sizeof(long))
        msgsnd(msgid, &message, sizeof(message.msg_text), 0);

        printf("Sender: Message sent.\n");
        
        // Wait a moment for child to receive before exiting
        sleep(1); 
    } 
    else {
        // --- RECEIVER PROCESS ---
        
        // Sleep to ensure sender has put something in the queue
        sleep(1);

        // msgrcv: Receives the message from the queue.
        // The '1' signifies we want to read messages of type 1.
        msgrcv(msgid, &message, sizeof(message.msg_text), 1, 0);

        printf("Receiver: Processed Data: %s", message.msg_text);

        // msgctl: Message Control. 
        // IPC_RMID: Removes the queue from the kernel immediately.
        msgctl(msgid, IPC_RMID, NULL);
        printf("Receiver: Queue destroyed.\n");
    }

    return 0;
}
